#Para instalar digite os seguintes comandos no terminal:

	cd /diretório_em_que_a_pasta_jogodavelha_está
	make
	sudo make install
	make clean

#Pronto, agora basta digitar

	jogodavelha

#para jogar.

#Para remover o jogo basta ir para o mesmo diretório pelo terminal e digitar

	sudo make remove

jogodavelha (se quiser remover completamente).
